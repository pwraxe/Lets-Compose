@OptIn(ExperimentalMaterial3Api::class)
@SuppressLint("SetJavaScriptEnabled")
@Composable
fun PlayGroundScreen(modifier: Modifier = Modifier) {
    Column (modifier = modifier
        .fillMaxSize()
        .background(color = Color.White)
        .padding(10.dp)) {

        Text(
            text = "This is first V text",
            fontFamily = fontMontserrat,
            fontSize = 20.sp
        )
        HorizontalDivider()
        Text(
            text = "This is second V text",
            fontFamily = fontMontserrat,
            fontSize = 20.sp
            )


        HorizontalDivider(
            modifier = modifier
                .padding(10.dp),
            thickness = 2.dp,
            color = Color.Black,
        )

        LazyColumn {
            items(5) {
                Text(text = "This is row no $it")
                HorizontalDivider()
            }
        }

        Spacer(modifier = modifier.padding(top = 10.dp))

        Row {
            Text(
                text = "This is first H text",
                fontFamily = fontMontserrat,
                fontSize = 20.sp
            )

            //VerticalDivider()
            VerticalDivider(
                modifier = modifier
                    .padding(
                            start = 10.dp,
                            end = 10.dp),
                thickness = 2.dp, color = Color.Black
            )
            Text(
                text = "This is second H text",
                fontFamily = fontMontserrat,
                fontSize = 20.sp
            )
        }
    }
}
