                NavigationBar(
                    modifier = modifier,
                    containerColor = Color.Yellow,
                    tonalElevation = 4.dp,
                ) {
                    NavigationBarItem(
                        selected = selectedMenu == 0,
                        onClick = { selectedMenu = 0 },

                            }
                        })


                    NavigationBarItem(
                        selected = selectedMenu == 2,
