@Preview
@Composable
fun PlayGroundPreview() {
    PlayGroundScreen()
}

@Composable
fun PlayGroundScreen(modifier: Modifier = Modifier) {

    Column(modifier = modifier
        .background(color = Color.White)
        .fillMaxSize()
        .padding(10.dp)

    ) {

        Row (verticalAlignment = Alignment.CenterVertically) {
            Switch(checked = true, onCheckedChange = {})
            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Default Switch - ON")
        }

        Row (verticalAlignment = Alignment.CenterVertically) {
            Switch(checked = false, onCheckedChange = {})
            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Default Switch - OFF")
        }

        HorizontalDivider(modifier = modifier.padding(top = 10.dp))

        var checked by remember { mutableStateOf(false) }

        Row (verticalAlignment = Alignment.CenterVertically) {
            Switch(
                checked = checked,
                onCheckedChange = { checked = it },
                thumbContent = {
                    if(checked) {
                        Icon(
                            imageVector = Icons.Outlined.Check,
                            contentDescription = "switch on")
                    } else {
                        Icon(
                            imageVector = Icons.Outlined.Clear,
                            contentDescription ="switch off")
                    }
                }
            )
            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Switch with icon - OFF")
        }
        Row (verticalAlignment = Alignment.CenterVertically) {
            Switch(
                checked = !checked,
                onCheckedChange = { checked = it },
                thumbContent = {
                    if(!checked) {
                        Icon(
                            imageVector = Icons.Outlined.Check,
                            contentDescription = "switch on")
                    } else {
                        Icon(
                            imageVector = Icons.Outlined.Clear,
                            contentDescription ="switch off")
                    }
                }
            )
            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Switch with icon - ON")
        }

        HorizontalDivider(modifier = modifier.padding(top = 10.dp))

        Row (verticalAlignment = Alignment.CenterVertically) {
            Switch(
                checked = checked,
                onCheckedChange = { checked = it },
                thumbContent = {
                    if(checked) {
                        Icon(
                            imageVector = Icons.Outlined.Check,
                            contentDescription = "switch on")
                    } else {
                        Icon(
                            imageVector = Icons.Outlined.Clear,
                            contentDescription ="switch off")
                    }
                },
                colors = SwitchDefaults.colors(
                    uncheckedBorderColor = Color.Black,
                    uncheckedIconColor = Color.Blue,
                    uncheckedThumbColor = Color.Yellow,
                    uncheckedTrackColor = Color.Green
                )
            )
            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Switch with icon and Color - OFF")
        }


        Row(verticalAlignment = Alignment.CenterVertically) {
            Switch(
                checked = !checked,
                onCheckedChange = { checked = it },
                thumbContent = {
                    if(checked) {
                        Icon(
                            imageVector = Icons.Outlined.Clear,
                            contentDescription ="switch off")
                    } else {
                        Icon(
                            imageVector = Icons.Outlined.Check,
                            contentDescription = "switch on")
                    }
                }, colors = SwitchDefaults.colors(
                    checkedBorderColor = Color.LightGray,
                    checkedIconColor = Color.Yellow,
                    checkedThumbColor = Color.Green,
                    checkedTrackColor = Color.Blue
                )
            )

            Spacer(modifier = modifier.padding(start = 10.dp))
            Text(text = "Switch with icon and Color - OFF")
        }
    }
}
